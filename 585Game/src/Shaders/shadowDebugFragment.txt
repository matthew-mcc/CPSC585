#version 330 core
out vec4 FragColor;
  
in vec2 TexCoords;

uniform sampler2D depthMap;
uniform bool isDepth;

void main()
{             
    float depthValue = texture(depthMap, TexCoords).r;
    if (isDepth) FragColor = vec4(vec3(depthValue), 1.0);
    else FragColor = texture(depthMap, TexCoords);
}
